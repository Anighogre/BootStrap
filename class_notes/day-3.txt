1) alert class
        1> alert alert-primary
        2> alert alert-secondary
        3> alert alert-danger
        4> alert alert-info
        5> alert alert-warning
        6> alert alert-success
        7> alert alert-light
        8> alert alert-dark

    2) alert-*(primary,secondary,danger,success,info,warning,light,dark)
        * ==> colors (8 colors are possible)


EX:
    <!-- Alert boxes -->

    <div class="alert alert-danger">  --> it will change the background color of the div
        <p>alert alert-danger</p>
    </div>
    
    <div class="alert alert-success">
        <p>alert alert-success</p>
    </div>

    <div class="alert alert-primary">
        <p>alert alert-primary</p>
    </div>

    <div class="alert alert-info">
        <p>alert alert-info</p>
    </div>

    <div class="alert alert-warning">
        <p>alert alert-warning</p>
    </div>

    <div class="alert alert-dark">
        <p>alert alert-dark</p>
    </div>

    <div class="alert alert-secondary">
        <p>alert alert-secondary</p>
    </div>

    <div class="alert alert-light">
        <p>alert alert-light</p>
    </div>
